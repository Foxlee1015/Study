            def __init__(self, first, last, pay):

                self.last = last
                self.email = first + '.' + last + '@noreply.com'
                self.pay = int(pay)

                Employee.number_of_emps +=1   # self.number_of_emps += 1

            def apply_raise(self):
                self.pay = self.pay * self.raise_amount


            @classmethod
