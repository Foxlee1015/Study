* URL : /board/xxxx // x 값을 article_id 로 대입


        @app.route("/board/<article_id>")
        def board(article_id):
            return "{}번 게시물".format(article_id)

    
* URL : /board      // 10번 게시물 return 
    
    
    
      @app.route("/board", defaults={ "article_id" : 10 })
      def board(article_id):
          return "{}번 게시물".format(article_id)
