        from rest_framework import serializers
        # 다른 앱의 모델 가져오기
        from django.apps import apps
        Patent = apps.get_model('main', 'Patent')   


        class PatentSerializer(serializers.ModelSerializer):
            class Meta:
                model = Patent
                fields = '__all__'
