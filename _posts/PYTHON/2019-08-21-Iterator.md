        class Counter:
            def __init__(self, stop):
                self.current = 0
                self.stop = stop


            # 반복가능한 객체로 만들어짐  이 메서드 없으면   For i in Counter(3):  error : TypeError: 'Counter' object is not iterable
            def __iter__(self):
                return self


            def __next__(self):
                if self.current < self.stop:
                    r = self.current
                    self.current += 1   # 다음 숫자 간격 조절 가능
                    return r
                else:
                    raise StopIteration

            # 리스트 인덱스처럼 서정
            def __getitem__(self, index):
                if index < self.stop:
                    return index
                else:
                    raise IndexError

        print("Counter(3)[1] : ", Counter(3)[1])

        for i in Counter(3):
            print(i)

        x = Counter(5)
        print(x.__next__())
        print(x.__next__())
        print(x.__next__())
        print(x.__next__())

        a,b,c, d, e = Counter(5)
        print(a, b, c, d, e)

        q, w, _, r = Counter(4)
        print(q, w, r)

        this = iter(range(3))
        print(next(this))
        print(next(this))
        print(next(this))


        import random
        for i in iter(lambda : random.randint(0,5), 2):
            print(i, end='')
