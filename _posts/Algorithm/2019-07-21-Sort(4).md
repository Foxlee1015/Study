        list = [3, 4, 6, 7, 8, 1, 2, 5, 9]
        n = len(list)


        # heapify
        for i in range(n):

            # c = children // root = patent
            c = i
            while c != 0:
                root = (c - 1) // 2
                print("c : ",c, "root : ", "list : ", list )
                # swap root and c if c > root  ( c, root = index of list )
                if list[root] < list[c]:
                    temp = list[root]
                    list[root] = list[c]
                    list[c] = temp

                c = root

        print(list)
